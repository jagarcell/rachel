-- Database: configdb

-- DROP DATABASE configdb;

CREATE DATABASE configdb
  WITH OWNER = postgres
       ENCODING = 'UTF8'
       TABLESPACE = pg_default
       LC_COLLATE = 'en_US.UTF-8'
       LC_CTYPE = 'en_US.UTF-8'
       CONNECTION LIMIT = -1;

-- Table: enlaces

-- DROP TABLE enlaces;

CREATE TABLE enlaces
(
  enlace character varying(100),
  comando character varying(500),
  description character varying(500),
  idunico bigserial NOT NULL,
  image character varying(500),
  title character varying(100),
  CONSTRAINT enlaces_pkey PRIMARY KEY (idunico)
)
WITH (
  OIDS=FALSE
);
